  \subsection{Modelo e dados do Dual}

  \begin{frame}{Modelo Primal}

    \begin{equation*}
     \textrm{Max.} Z = \sum_j^n c_jx_j \hspace{1.5cm} \textrm{sujeito a}
    \end{equation*}

    \begin{equation*}
     \sum_{j=0}^n a_{ij}x_j \leq b_i  \hspace{1.5cm} (i = 1, 2, 3..., m)
    \end{equation*}

    \begin{equation*}
     x_j \geq 0 \hspace{1.5cm} (j = 1, 2, 3..., n)
    \end{equation*}

  \end{frame}


  \begin{frame}{Obtendo o dual}
    \begin{itemize}[<+->]
      \item
        Função objetivo do primal é  maximização, então a do dual é minimização.

      \item
        Termos constantes das restrições do dual são os coeficientes da função objetiva do primal e vice-versa.

      \item
        O número de incógnitas do dual ($m$ valores de $y_i$) é igual ao número de restrições do primal.

      \item
        O número de restrições do dual é igual ao número de incógnitas do primal ($m$ valores de $x_j$).

      \item
        A matriz de coeficientes do dual é a transposta da matriz de coeficientes do primal.
    \end{itemize}


  \end{frame}

  \begin{frame}{Modelo Dual}

    \begin{equation*}
     \textrm{Min.} D = \sum_i^m b_iy_j \hspace{1.5cm} \textrm{sujeito a}
    \end{equation*}

    \begin{equation*}
     \sum_{i=0}^m a_{ij}y_i \geq c_j  \hspace{1.5cm} (j = 1, 2, 3..., n)
    \end{equation*}

    \begin{equation*}
     y_i \geq 0 \hspace{1.5cm} (i = 1, 2, 3..., m)
    \end{equation*}

  \end{frame}


  \begin{frame}{Problema Dual}
    Função objetivo:
    \begin{equation*}
    \textrm{Min.}  D = 100 \times y_1 - 50 \times y_2 + 50 \times y_3 + 35 \times y4 + 50 \times y_5 + 0 \times y_6
    \end{equation*}
    Restrições:
    \begin{tabular}{ccccccc}
      $y1$ & $-y2$ & $+ y3$ & & & & $\geq$ 70\\
      $y1$ &  & & $- y4$&  & $+ y6$& $\geq$ 104\\
      $y1$ &  & & & $+ y5$ & $-4 y6$& $\geq$ 80\\
    \end{tabular}
  \end{frame}
